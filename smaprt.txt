# ======================================================
# FUNCTIONS
# ======================================================

def sentiment_analysis():

    positive = random.randint(65,80)
    neutral = random.randint(10,20)

    negative = 100 - (positive + neutral)

    return positive, neutral, negative


def offer_prediction(price):

    current_discount = random.randint(5,15)

    next_discount = random.randint(10,25)

    current_price = int(
        price - (price * current_discount / 100)
    )

    future_price = int(
        price - (price * next_discount / 100)
    )

    return (
        current_discount,
        next_discount,
        current_price,
        future_price
    )


def category(price):

    if price < 1000:
        return "Budget Product"

    elif price < 5000:
        return "Mid Range Product"

    else:
        return "Premium Product"


# ======================================================
# MAIN ROUTE
# ======================================================

@app.route("/", methods=["GET","POST"])

def home():

    data = {}

    if request.method == "POST":

        name = request.form["name"]

        model = request.form["model"]

        price = int(request.form["price"])

        appname = request.form["app"]

        positive, neutral, negative = sentiment_analysis()

        (
            current_discount,
            next_discount,
            current_price,
            future_price

        ) = offer_prediction(price)

        product_type = category(price)

        rating = round((positive/100)*5,1)

        if next_discount > current_discount:

            suggestion = "Wait For Upcoming Sale"

            season = "Festival Sale (Oct-Nov)"

        else:

            suggestion = "Buy Now"

            season = "Current Offer Season"

        data = {

            "name": name,
            "model": model,
            "price": price,
            "app": appname,

            "category": product_type,

            "current_discount": current_discount,

            "next_discount": next_discount,

            "current_price": current_price,

            "future_price": future_price,

            "positive": positive,

            "neutral": neutral,

            "negative": negative,

            "rating": rating,

            "season": season,

            "suggestion": suggestion

        }

    return render_template_string(
        HTML,
        data=data
    )
